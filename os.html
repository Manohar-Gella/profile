C Code Example


#include <stdio.h>
#include <stdlib.h>
#include <errno.h>
#include <sys/types.h>
#include <unistd.h>
#define MAXLINE 16

int main(){
    int fd[2], n;
    pid_t pid;
    char line[MAXLINE];
    if(pipe(fd) < 0){
        perror("Pipe error");
    }
    if((pid = fork()) < 0){
        perror("fork error");
    }
    else if (pid>0)
    {
        close(fd[0]);
        write(fd[1], "Hello", 10);
    }
    else{
        close(fd[1]);
        n = read(fd[0], line, MAXLINE);
        write(STDOUT_FILENO, line, n);
    }
    exit(0);
}
