Shell Scripts

Directory Management

echo "Directory Management"
mkdir mydir
echo "Directory 'mydir' created."

ls -l

rmdir mydir
echo "Directory 'mydir' deleted."

File Management

echo "File Management"
touch myfile.txt
echo "File 'myfile.txt' created."

ls -l myfile.txt

rm myfile.txt
echo "File 'myfile.txt' deleted."

Weekday Loop

echo "Weekday Loop"
i=1
weekday="mon tue wed thur fri sat"
for day in $weekday
do
echo "Weekday" $((i++)): "$day"
done
